Custom Filters

Como criar seus próprios filters?

Sintaxe de um custom filter

Formatação de data usando moment.js

angular.module('myApp', []).filter('myFormatDate', function() {
  return function(input) {
     if (typeof moment === 'function') { 
        if( typeof input !== 'undefined' && moment(input).isValid() ) {
           input = moment(input).format('MM/DD/YYYY [at] h:mm A');
        }
     }
     return input;
  };
})

Sintaxe de um custom filter

No html:

<article>
   <h1>{{ article.title }}</h1>
   <time>{{ article.date | myFormatDate }}</time>
</article>

Recebendo parâmetros

Formatação de data usando moment.js

angular.module('myApp', []).filter('myFormatDate', function( stringFormat ) {
  return function(input) {
     if (typeof moment === 'function') { 
        if( typeof input !== 'undefined' && moment(input).isValid() ) {
           input = moment(input).format( stringFormat );
        }
     }
     return input;
  };
})

Recebendo parâmetros

No html:

<article>
   <h1>{{ article.title }}</h1>
   <time>{{ article.date | myFormatDate:'MM/DD/YYYY [at] h:mm A' }}</time>
</article>
